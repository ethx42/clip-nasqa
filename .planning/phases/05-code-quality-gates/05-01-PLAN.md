Step 3 — Create `lint-staged.config.js` at root (ESM format — root package.json has no "type": "module" so this needs `.js` extension with `export default`; actually check if root package.json has `"type": "module"` — if not, use CommonJS `module.exports =` syntax instead):
```js
// lint-staged.config.js
export default {
  "packages/frontend/**/*.{ts,tsx}": [
    "eslint --fix --config packages/frontend/eslint.config.mjs --no-warn-ignored",
    "prettier --write",
  ],
  "packages/{core,functions}/**/*.ts": ["prettier --write"],
  "**/*.{json,md,css,yml}": ["prettier --write"],
};
```
IMPORTANT: Check root package.json for `"type"` field. If absent (CommonJS default), rename to `lint-staged.config.mjs` or add `"type": "module"` — research shows ESM syntax. Use `.mjs` extension to be safe without modifying package.json type field.

Step 4 — Update `packages/frontend/eslint.config.mjs` to add `eslint-config-prettier/flat` as the LAST entry:
```js
import { defineConfig, globalIgnores } from "eslint/config";
import nextVitals from "eslint-config-next/core-web-vitals";
import nextTs from "eslint-config-next/typescript";
import eslintConfigPrettier from "eslint-config-prettier/flat";

const eslintConfig = defineConfig([
  ...nextVitals,
  ...nextTs,
  globalIgnores([
    ".next/**",
    "out/**",
    "build/**",
    "next-env.d.ts",
  ]),
  eslintConfigPrettier,  // MUST be last — disables conflicting formatting rules
]);

export default eslintConfig;
```
